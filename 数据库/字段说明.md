--uesrs(用户信息表)--

字段名                    类型        约束 / 默认值                 说明

phone                    TEXT        PRIMARY KEY                  用户手机号（系统主要账号标识）

student_id               TEXT        DEFAULT ''                   学号 (12位)

name                     TEXT        NOT NULL                     真实姓名或昵称

password                 TEXT        NOT NULL                     SHA-256 加密后的密码哈希值
  
college                  TEXT        DEFAULT ''                   学院
 
major                    TEXT        DEFAULT ''                   专业

class_name               TEXT        DEFAULT ''                   班级

qq                       TEXT        DEFAULT ''                   QQ 联系方式

wechat                   TEXT        DEFAULT ''                   微信 联系方式
 
skills                   TEXT        DEFAULT ''                   技能标签 (逗号分隔)

honors                   TEXT        DEFAULT ''                   个人荣誉

bio                      TEXT        DEFAULT ''                   个人简介 / 个性签名

is_first_login           INTEGER     DEFAULT 1                    "是否首次登录 (1=是, 0=否)，用于强制引导填资料"

created_at               TIMESTAMP   DEFAULT CURRENT_TIMESTAMP    账号创建时间

--projects (招募项目表)--

字段名                    类型        约束 / 默认值                 说明

id                       INTEGER     PRIMARY KEY AUTOINCREMENT    项目唯一自增ID

title                    TEXT        NOT NULL                     项目招募标题

leader_phone             TEXT        FOREIGN KEY                  队长手机号 (关联 users 表)

description              TEXT        无                           项目详细描述 (支持 Markdown)

tags                     TEXT        无                           技能需求标签 (逗号分隔)

base_members             INTEGER     DEFAULT 1                    队伍已有人数

required_members         INTEGER     DEFAULT 3                    队伍需要达到的总人数

status                   TEXT        DEFAULT '招募中'             "项目状态 ('招募中', '已截止')"

is_deleted               INTEGER     DEFAULT 0                    逻辑删除标记 (1=已删除)

is_hidden                INTEGER     DEFAULT 0                    隐藏标记 (1=仅队长可见)

--applications (项目申请表)--

字段名                   类型         约束 / 默认值                 说明

id                      INTEGER      PRIMARY KEY AUTOINCREMENT    申请记录唯一自增ID

proj_id                 INTEGER      FOREIGN KEY                  申请的项目ID

applicant_phone         TEXT         FOREIGN KEY                  申请人手机号

status                  TEXT         DEFAULT '待审核'             "状态 ('待审核', '已同意', '已拒绝', '已取消', '已移出')"

applicant_visible       INTEGER      DEFAULT 1                    申请人端是否可见

leader_visible          INTEGER      DEFAULT 1                    队长端是否可见

leader_read             INTEGER      DEFAULT 0                    队长是否已读红点标记

applicant_read          INTEGER      DEFAULT 0                    申请人是否已读红点标记

--messages (消息记录表)--

字段名                  类型          约束 / 默认值                 说明

id                     INTEGER       PRIMARY KEY AUTOINCREMENT    消息唯一自增ID

sender_phone           TEXT          NOT NULL                     发送者手机号 ('system' 代表系统通知)

chat_type              TEXT          NOT NULL                     聊天类型 ('group' 或 'private')

target_id              TEXT          NOT NULL                     目标ID (群ID或对方手机号)

content                TEXT          NOT NULL                     消息具体内容

created_at             TIMESTAMP     DEFAULT CURRENT_TIMESTAMP    消息发送时间

--chat_state (聊天游标状态表)--

字段名                 类型          约束 / 默认值                  说明

phone                 TEXT          PRIMARY KEY 1                 用户手机号

chat_type             TEXT          PRIMARY KEY 2                 聊天类型

target_id             TEXT          PRIMARY KEY 3                 目标ID

last_read_msg_id      INTEGER       DEFAULT 0                     最后读取的消息ID (用于未读统计)

cleared_up_to_msg_id  INTEGER       DEFAULT 0                     清空记录的游标ID
